import jwt import datetime SECRET_KEY = "your-secret-key" # Create (encode) a token payload = { "user_id": 123, "username": "alice", "exp": datetime.datetime.utcnow() + datetime.timedelta(hours=1) } token = jwt.encode(payload, SECRET_KEY, algorithm="HS256") print("Token:", token) # Decode (verify) a token try: decoded = jwt.decode(token, SECRET_KEY, algorithms=["HS256"]) print("Decoded:", decoded) except jwt.ExpiredSignatureError: print("Token has expired") except jwt.InvalidTokenError: print("Invalid token")
